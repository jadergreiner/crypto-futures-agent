```python
        Mapeia R1: exponential backoff (1s, 2s, 4s)
```

```python
# core/model2/io_retry.py

import functools
import time
import signal
import logging
import tempfile
import os
from pathlib import Path
from contextlib import contextmanager
from typing import Optional, Callable, Any, Dict

logger = logging.getLogger(__name__)

class IoRetryError(Exception):
    """Exceção customizada para I/O retry failures."""
    pass


def retry_with_backoff(
    retries: int = 3,
    backoff_seconds: tuple = (1, 2, 4, 8),
    fail_safe: bool = False
) -> Callable:
    """
    Decorator para retry com exponential backoff.

    Args:
        retries: número máximo de tentativas
        backoff_seconds: tuple de delays entre tentativas (ex: (1, 2, 4))
        fail_safe: se True, retorna False em vez de lançar exceção

    Returns:
        Função decorada com retry wrapper
    """
    def decorator(func: Callable) -> Callable:
        @functools.wraps(func)
        def wrapper(*args, **kwargs) -> Any:
            last_exception = None

            for attempt in range(1, retries + 1):
                try:
                    return func(*args, **kwargs)
                except (IOError, OSError, TimeoutError) as e:
                    last_exception = e
                    if attempt < retries:
                        delay = backoff_seconds[attempt - 1] if attempt - 1 < len(backoff_seconds) else backoff_seconds[-1]
                        logger.warning(
                            f"Retry attempt {attempt}/{retries} failed: {str(e)}. "
                            f"Waiting {delay}s before next attempt..."
                        )
                        time.sleep(delay)
                    else:
                        logger.error(f"All {retries} retries exhausted for {func.__name__}: {str(e)}")

            if fail_safe:
                return False
            raise IoRetryError(f"Failed after {retries} attempts: {str(last_exception)}")

        return wrapper
    return decorator


@contextmanager
def atomic_file_write(file_path: str, mode: str = 'w', encoding: str = 'utf-8'):
    """
    Context manager para escrita atômica (temp + rename).

    Garante que o arquivo final NUNCA fica em estado parcial.
    Se houver erro durante dump, arquivo original permanece inalterado.

    Args:
        file_path: path do arquivo final desejado
        mode: modo de escrita (padrão 'w')
        encoding: encoding (padrão 'utf-8')

    Yields:
        File object para escrita
    """
    temp_fd = None
    temp_path = None

    try:
        # Criar arquivo temporário no mesmo diretório (para garantir mesmo filesystem)
        tmpdir = Path(file_path).parent
        tmpdir.mkdir(parents=True, exist_ok=True)

        temp_fd, temp_path = tempfile.mkstemp(
            dir=str(tmpdir),
            prefix=f"{Path(file_path).stem}.",
            suffix=".tmp"
        )

        with os.fdopen(temp_fd, mode, encoding=encoding) as f:
            yield f

        # Se chegou aqui, escrita foi bem-sucedida
        # Rename atômico (Windows + Linux)
        os.replace(temp_path, file_path)
        logger.debug(f"Atomic write completed: {file_path}")

    except Exception as e:
        # Erro durante dump — arquivo temp é descartado, original inalterado
        if temp_path and os.path.exists(temp_path):
            os.remove(temp_path)
            logger.warning(f"Atomic write failed, temp file cleaned: {e}")
        raise


def read_json_with_retry(
    file_path: str,
    timeout_seconds: Optional[float] = 5.0,
    retries: int = 3,
    fail_safe: bool = False
) -> Optional[Dict]:
    """
    Ler JSON com retry + timeout.

    Args:
        file_path: path do arquivo JSON
        timeout_seconds: timeout total (5s padrão para reads)
        retries: número de tentativas
        fail_safe: retorna False em vez de lançar

    Returns:
        Dict com dados JSON, ou False em fail_safe, ou lança exceção
    """
    backoff = (1, 2, 2) if retries == 3 else (1,) * retries

    @retry_with_backoff(retries=retries, backoff_seconds=backoff, fail_safe=fail_safe)
    def _read():
        start = time.time()
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                elapsed = time.time() - start
                if timeout_seconds and elapsed > timeout_seconds:
                    raise TimeoutError(f"Read timeout exceeded ({elapsed:.1f}s > {timeout_seconds}s)")
                return json.load(f)
        except (FileNotFoundError, json.JSONDecodeError) as e:
            if fail_safe:
                return False
            raise

    return _read()


def write_json_with_retry(
    data: Dict,
    file_path: str,
    timeout_seconds: Optional[float] = 10.0,
    retries: int = 4,
    fail_safe: bool = False
) -> Optional[bool]:
    """
    Escrever JSON com retry + atomicidade + timeout.

    Args:
        data: dicionário a ser escrito
        file_path: path do arquivo JSON
        timeout_seconds: timeout total (10s padrão para writes)
        retries: número de tentativas
        fail_safe: retorna False em vez de lançar

    Returns:
        True se sucesso, False em fail_safe, ou lança exceção
    """
    backoff = (1, 2, 4, 3) if retries == 4 else (1,) * retries

    @retry_with_backoff(retries=retries, backoff_seconds=backoff, fail_safe=fail_safe)
    def _write():
        start = time.time()
        try:
            with atomic_file_write(file_path) as f:
                json.dump(data, f, indent=2, ensure_ascii=False)
                elapsed = time.time() - start
                if timeout_seconds and elapsed > timeout_seconds:
                    raise TimeoutError(f"Write timeout exceeded ({elapsed:.1f}s > {timeout_seconds}s)")
            return True
        except Exception as e:
            if fail_safe:
                return False
            raise

    return _write()
```
